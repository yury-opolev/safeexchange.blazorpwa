```csharp
/// <summary>
/// SecretAuditDirectionTests — pure-unit checks for the "reverse around merger"
/// strategy used by SafeExchangeSecretAudit when direction=desc. Ensures merged
/// ContentRead ranges are equivalent (same seqFrom/seqTo/chunkIds) regardless
/// of whether the merger received the page ascending or reversed.
/// </summary>

namespace SafeExchange.Tests
{
    using NUnit.Framework;
    using SafeExchange.Core;
    using SafeExchange.Core.Audit;
    using SafeExchange.Core.Model;
    using SafeExchange.Core.Model.Dto.Output;
    using System.Collections.Generic;
    using System.Linq;

    [TestFixture]
    public class SecretAuditDirectionTests
    {
        private static SecretAuditEvent Read(long seq, string actor, string content, string chunk)
        {
            var payload = DefaultJsonSerializer.Serialize(new { contentId = content, chunkId = chunk });
            return new SecretAuditEvent
            {
                id = SecretAuditEvent.MakeId("inst", seq),
                AuditInstanceId = "inst",
                SequenceNumber = seq,
                EventType = SecretAuditEventType.ContentRead,
                OccurredAt = new System.DateTime(2026, 1, 1, 0, 0, (int)seq, System.DateTimeKind.Utc),
                ActorSubjectType = SubjectType.User,
                ActorSubjectId = actor,
                ActorDisplayName = actor,
                Payload = payload,
                PrevHash = string.Empty,
                Hash = "h" + seq,
            };
        }

        // Mirrors the in-handler reverse-merge-reverse path. Kept in tests because the
        // production code does the same inline; this test pins the resulting shape.
        private static List<SecretAuditEventOutput> ReverseMergeReverse(IList<SecretAuditEvent> desc)
        {
            var asc = new List<SecretAuditEvent>(desc);
            asc.Reverse();
            var merged = ContentReadMerger.Merge(asc);
            merged.Reverse();
            return merged;
        }

        [Test]
        public void DescPath_MergesContentRunsLikeAsc_OnlyReversedInOutput()
        {
            var asc = new List<SecretAuditEvent>
            {
                Read(1, "a@x", "c1", "k1"),
                Read(2, "a@x", "c1", "k2"),
                Read(3, "a@x", "c1", "k3"),
            };
            var ascMerged = ContentReadMerger.Merge(asc);

            var desc = new List<SecretAuditEvent>(asc);
            desc.Reverse();
            var descMerged = ReverseMergeReverse(desc);

            Assert.That(descMerged.Count, Is.EqualTo(ascMerged.Count));
            Assert.That(descMerged[0].SequenceFrom, Is.EqualTo(ascMerged[0].SequenceFrom));
            Assert.That(descMerged[0].SequenceTo, Is.EqualTo(ascMerged[0].SequenceTo));
            Assert.That(descMerged[0].ChunkIds, Is.EquivalentTo(ascMerged[0].ChunkIds!));
        }

        [Test]
        public void DescPath_PreservesNewestFirstAcrossMixedTypes()
        {
            var asc = new List<SecretAuditEvent>
            {
                Read(10, "a@x", "c1", "k1"),
                Read(11, "a@x", "c1", "k2"),
                Read(12, "b@x", "c1", "k1"),
            };
            var desc = new List<SecretAuditEvent>(asc);
            desc.Reverse();
            var descMerged = ReverseMergeReverse(desc);

            // The first output row should be the newest one (the b@x single read).
            Assert.That(descMerged.First().Actor.SubjectId, Is.EqualTo("b@x"));
            Assert.That(descMerged.First().SequenceFrom, Is.EqualTo(12));

            // The second row should be the merged a@x run (seq 10..11).
            Assert.That(descMerged.Last().Actor.SubjectId, Is.EqualTo("a@x"));
            Assert.That(descMerged.Last().SequenceFrom, Is.EqualTo(10));
            Assert.That(descMerged.Last().SequenceTo, Is.EqualTo(11));
        }
    }
}
```

```razor
@code {

    [Parameter]
    public string ObjectName { get; set; }

    private bool loaded;

    private bool loading;

    private bool auditEnabled;

    private string? nextContinuation;

    private bool rawReads;

    private DateTime? filterFrom;

    private DateTime? filterTo;

    private string filterActor = string.Empty;

    private readonly List<SecretAuditEventOutput> allEvents = new();

    private List<SecretAuditEventOutput> visibleEvents = new();

    private readonly HashSet<string> expanded = new();

    private NotificationData? Notification;

    private readonly string[] knownEventTypes = new[]
    {
        "SecretCreated", "SecretMetadataUpdated", "SecretDeleted",
        "PermissionGranted", "PermissionRevoked",
        "ContentRead", "ContentWritten", "ContentCommitted",
        "AccessRequested", "AccessRequestApproved", "AccessRequestDenied",
    };

    private Dictionary<string, bool> eventTypeFilters = new();

    protected override async Task OnInitializedAsync()
    {
        this.StateContainer.IsInProgress = false;
        this.StateContainer.TakeNotification();

        this.ObjectName ??= string.Empty;
        this.StateContainer.SetCurrentPageHeader($"Audit log — '{this.ObjectName}'");

        foreach (var t in this.knownEventTypes)
        {
            this.eventTypeFilters[t] = true;
        }

        await this.LoadFirstPageAsync();

        await this.Telemetry.TrackEventAsync("AuditLogViewed", new Dictionary<string, string>
        {
            ["name"] = this.ObjectName,
        });
    }

    protected override async Task OnAfterRenderAsync(bool firstRender)
    {
        await TooltipsInitializerInstance.InitializeAllPlainTooltipsAsync();
    }

    private async Task LoadFirstPageAsync()
    {
        this.loaded = false;
        this.allEvents.Clear();
        this.expanded.Clear();
        this.nextContinuation = null;

        var resp = await this.apiClient.GetSecretAuditAsync(
            this.ObjectName,
            direction: "desc",
            from: this.filterFrom,
            to: this.filterTo,
            pageSize: 100,
            continuation: null,
            raw: this.rawReads);

        if (!"ok".Equals(resp.Status))
        {
            this.Notification = new NotificationData
            {
                Type = NotificationType.Warning,
                Status = resp.Status,
                Message = $"Cannot load audit log: {(resp.Error ?? "Unknown exception.")}.",
            };
            this.auditEnabled = false;
            this.loaded = true;
            await this.Telemetry.TrackEventAsync("AuditLogLoadFailed", new Dictionary<string, string>
            {
                ["name"] = this.ObjectName,
                ["status"] = resp.Status ?? string.Empty,
            });
            return;
        }

        var page = resp.Result ?? new SecretAuditPageOutput();
        this.auditEnabled = page.AuditEnabled;
        this.allEvents.AddRange(page.Events);
        this.nextContinuation = page.NextContinuation;
        this.RecomputeVisible();
        this.loaded = true;
    }

    private async Task LoadMoreAsync()
    {
        if (string.IsNullOrEmpty(this.nextContinuation))
        {
            return;
        }
        this.loading = true;
        try
        {
            var resp = await this.apiClient.GetSecretAuditAsync(
                this.ObjectName,
                direction: "desc",
                from: this.filterFrom,
                to: this.filterTo,
                pageSize: 100,
                continuation: this.nextContinuation,
                raw: this.rawReads);

            if (!"ok".Equals(resp.Status))
            {
                this.Notification = new NotificationData
                {
                    Type = NotificationType.Warning,
                    Status = resp.Status,
                    Message = $"Cannot load older events: {(resp.Error ?? "Unknown exception.")}.",
                };
                return;
            }

            var page = resp.Result ?? new SecretAuditPageOutput();
            this.allEvents.AddRange(page.Events);
            this.nextContinuation = page.NextContinuation;
            this.RecomputeVisible();
        }
        finally
        {
            this.loading = false;
        }
    }

    private async Task ApplyServerFiltersAsync() => await this.LoadFirstPageAsync();

    private async Task ResetFiltersAsync()
    {
        this.filterFrom = null;
        this.filterTo = null;
        this.filterActor = string.Empty;
        foreach (var t in this.knownEventTypes)
        {
            this.eventTypeFilters[t] = true;
        }
        this.rawReads = false;
        await this.LoadFirstPageAsync();
    }

    private void OnToggleEventType(string type, bool value)
    {
        this.eventTypeFilters[type] = value;
        this.RecomputeVisible();
    }

    private async Task OnToggleRaw(bool value)
    {
        this.rawReads = value;
        await this.LoadFirstPageAsync();
    }

    private void ToggleRow(string key)
    {
        if (!this.expanded.Remove(key))
        {
            this.expanded.Add(key);
        }
    }

    private void RecomputeVisible()
    {
        IEnumerable<SecretAuditEventOutput> q = this.allEvents;
        q = q.Where(e => this.eventTypeFilters.TryGetValue(e.EventType, out var on) ? on : true);
        if (!string.IsNullOrWhiteSpace(this.filterActor))
        {
            var needle = this.filterActor.Trim();
            q = q.Where(e =>
                (e.Actor.SubjectName?.Contains(needle, StringComparison.OrdinalIgnoreCase) ?? false)
                || (e.Actor.SubjectId?.Contains(needle, StringComparison.OrdinalIgnoreCase) ?? false));
        }
        this.visibleEvents = q.ToList();
    }

    private void DismissNotification()
    {
        this.Notification = null;
    }

    private static string FormatTime(SecretAuditEventOutput ev)
    {
        var ts = ev.FirstAt ?? ev.OccurredAt;
        return ts.ToUniversalTime().ToString("yyyy-MM-dd HH:mm:ssZ");
    }

    private static string BadgeClassForType(string type) => type switch
    {
        "SecretCreated" or "SecretMetadataUpdated" or "SecretDeleted" => "text-bg-secondary",
        "PermissionGranted" or "PermissionRevoked" => "text-bg-primary",
        "ContentRead" or "ContentWritten" or "ContentCommitted" => "text-bg-info",
        "AccessRequested" or "AccessRequestApproved" or "AccessRequestDenied" => "text-bg-warning",
        _ => "text-bg-secondary",
    };

    private static string FormatSeq(SecretAuditEventOutput ev)
    {
        if (ev.SequenceFrom.HasValue && ev.SequenceTo.HasValue && ev.SequenceFrom != ev.SequenceTo)
        {
            return $"{ev.SequenceFrom}–{ev.SequenceTo}";
        }
        return ev.SequenceNumber.ToString();
    }

    private static string Truncate(string s)
    {
        if (string.IsNullOrEmpty(s))
        {
            return string.Empty;
        }
        return s.Length > 12 ? s.Substring(0, 12) + "…" : s;
    }

    private static string PrettyPayload(SecretAuditEventOutput ev)
    {
        if (ev.Payload is null)
        {
            return "{}";
        }
        try
        {
            return JsonSerializer.Serialize(ev.Payload, new JsonSerializerOptions { WriteIndented = true });
        }
        catch
        {
            return ev.Payload.ToString() ?? "{}";
        }
    }

    private static string SummariseEvent(SecretAuditEventOutput ev)
    {
        switch (ev.EventType)
        {
            case "ContentRead":
                if (ev.ChunkIds is { Count: > 0 })
                {
                    return $"{ev.ContentId} · {ev.ChunkIds.Count} chunks";
                }
                return $"{ev.ContentId ?? string.Empty}";
            case "ContentWritten":
                return $"{ev.ContentId ?? string.Empty}";
            case "ContentCommitted":
                return ev.ContentId ?? string.Empty;
            default:
                return string.Empty;
        }
    }
```
